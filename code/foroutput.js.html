01 import { mount } from '@vue/test-utils'
02 import { defineComponent } from 'vue'
03 
04 const BaseButton = defineComponent({
05   name: 'BaseButton',
06   template: `
07     <button><slot></slot></button>
08   `
09 })
10 
11 const BaseModal = defineComponent({
12   name: 'BaseModal',
13   components: {
14     BaseButton
15   },
16   template: `
17         <div>
18             <BaseButton>確認</BaseButton>
19         </div>
20     `
21 })
22 
23 it('使用 Component Name 載入', () => {
24   const wrapper = mount(BaseModal)
25   console.log(wrapper.findComponent({ name: 'BaseButton' }).html())
26   // <BaseButton>確認</BaseButton>
27 })
28 
29 it('直接引入元件', () => {
30   const wrapper = mount(BaseModal)
31   console.log(wrapper.findComponent(BaseButton).html())
32   // <BaseButton>確認</BaseButton>
33 })
