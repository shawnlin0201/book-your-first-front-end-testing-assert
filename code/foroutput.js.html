01 const formatDate = (date) => {
02   var newDate = new Date(date)
03   var offset = new Date().getTimezoneOffset()
04   newDate = new Date(newDate.getTime() - offset * 60 * 1000)
05 
06   return newDate.toISOString().split('T')[0]
07 }
08 it('formatDate 應該針對日期輸出對應的格式', async () => {
09   expect(formatDate('2023/07/22')).toBe('2023-07-22')
10 })
